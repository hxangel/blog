function encodeFiles($fileName)
{

function tree($directory)
{
    $mydir = dir($directory);
    while ($file = $mydir->read()) {
        if ((is_dir("$directory/$file")) AND ($file != ".") AND ($file != "..")) {
            tree("$directory/$file");
        } else {
            $file ="$directory/$file";

//            echo "<li>$file</li>\n";
            if(is_file($file)){
                encodeFiles($file);
            }
        }
    }
}
